一、创建版本库
    mkdir MyGit   //创建目录
    cd MyGit      //进入该目录
    pwd           //显示当当前目录
    git init      //把当前目录变成版本库
    ls -ah        //显示版本库中.git目录

二、文件的管理
    git status    //查看版本库当前状态
    cat readme.txt  //查看文件内容
    git diff readme.txt //查看文件修改情况（未add）
    git add readme.txt|git add -A  //把文件放入stage
    git commit -m "add readme.txt" //把stage中的文件提交

    git log|git log --pretty=oneline //查看提交日志
    git reflog   //记录每一次命令
    
    git reset --hard HEAD^|git reset --hard HEAD~10  //会退到之前的版本
    git reset --hard 3628164  //会退到指定版本
    
    git checkout -- readme.txt  //1、没有放到暂存区，撤销到和版本库一样
                                //2、添加到暂存区，撤回到添加到暂存区之前的状态
    git reset|git reset HEAD   //把暂存区的所有内容撤销

    git rm test.txt     //从工作区删除test文件
    git commit -m "remove test.txt" //删除之后记得提交

四、分支与合并
    git branch                  //查看现有分支，*指向为当前分支
    git branch addcommand       //创建分支
    git checkout addcommand     //切换分支
    git checkout -b addcommand  //创建+切换分支    
    git merge addcommand        //合并某分支到当前分支
    git branch -d addcommand    //删除分支
